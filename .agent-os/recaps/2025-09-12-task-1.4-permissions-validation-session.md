### Permission Resolution Algorithm
**Core Resolution Logic**:
```typescript
export async function checkPermission(query: PermissionQuery): Promise<PermissionResult> {
  const permissions = await loadPermissions();
  const rules = permissions.rules || [];
  
  // Find first matching rule (precedence)
  const matchingRule = rules.find(rule => matchesPattern(query, rule));
  
  if (matchingRule) {
    return { allowed: matchingRule.action === 'allow', rule: matchingRule };
  }
  
  // Fallback to defaults
  return { allowed: false, rule: null };
}
```
